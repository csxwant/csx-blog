              # 有序数组的平方open in new window

              # 解决方法

              # 1. 对负数取反后再排序

              # 3. 双指针

              # 颜色分类open in new window

              给定一个包含红色、白色和蓝色、共n 个元素的数组nums，原地对它们进行排序，使得相同颜色的元素相邻，并按照红色、白色、蓝色顺序排列。

              我们使用整数 0、1 和 2 分别表示红色、白色和蓝色。

              必须在不使用库的sort函数的情况下解决这个问题。

              # 解决方法

              # 1. 单指针

              先将所有0元素插到最前方，再将所有1元素插入到0元素后

              # 2. 双指针

              • p0 p1
                对0与1进行排序
              • p0 p2
                对0与2进行排序

              # 三数之和open in new window

              给你一个包含 n 个整数的数组nums，判断nums中是否存在三个元素 a，b，c ，使得
              a + b + c = 0 ？请你找出所有和为 0 且不重复的三元组。

              注意：答案中不可以包含重复的三元组。

              # 解决方案

              一层循环确定第一个元素，双指针确定其他两个元素，过滤掉无解的情况：

              • 最小和都大于target
              • 最大和都小于targer
              class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        if(nums==null||nums.length<3){
            return new ArrayList<>();
        }
        int len=nums.length;
        // 排序
        Arrays.sort(nums);
        List<List<Integer>> result = new ArrayList<>();

        // 遍历选取第一个元素
        for(int i=0; i<len-2; i++){
            // 去掉重复情况
            if(i>0&&nums[i]==nums[i-1]){
                continue;
            }

            // 去掉必无解的情况
            // 最小和已经大于0
            if(nums[i]+nums[i+1]+nums[i+2]>0){
                break;
            }
            // 最大和也小于0
            if(nums[i]+nums[len-1]+nums[len-2]<0){
                continue;
            }

            // 双指针选取另外两个元素
            int left=i+1, right=len-1;
            while(left<right){
                int sum=nums[i]+nums[left]+nums[right];
                // 满足条件，加入
                if(sum==0){
                    result.add(Arrays.asList(nums[i], nums[left], nums[right]));

                    // 移动指针，并去掉重复情况
                    left++;
                    right--;
                    while(left<right&&nums[left]==nums[left-1]){
                        left++;
                    }
                    while(left<right&&nums[right]==nums[right+1]){
                        right--;
                    }
                }
                // 小于0
                else if(sum<0){
                    left++;
                }
                else{
                    right--;
                }
            }
        }
        return result;
    }
}

              # 四数之和open in new window

              给你一个由 n 个整数组成的数组nums ，和一个目标值 target 。请你找
              出并返回满足下述全部条件且不重复的四元组[nums[a], nums[b], nums[c], nums[d]]
              （若两个四元组元素一一对应，则认为两个四元组重复）：

              0 <= a, b, c, d< n a、b、c 和 d 互不相同 nums[a] + nums[b] + nums[c] + nums[d] == target 你可以按 任意顺序 返回答案 。

              # 解决方案

              两层循环确定前两个元素，双指针确定其他两个元素，过滤掉无解情况
              与三数之和类似

              class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        if(nums.length<4){
            return new ArrayList<>();
        }
        List<List<Integer>> result = new ArrayList<>();
        // 先进行排序
        Arrays.sort(nums);
        int len=nums.length;
        // 加法判断时间溢出
        // 第一重循环确定第一个元素
        for(int i=0; i<len-3; i++){
            // 如果与前一个元素值一样，则跳过，避免重复
            if(i>0&&nums[i]==nums[i-1]){
                continue;
            }

            // 最小的和已经超过target
            if((long)nums[i]+nums[i+1]+nums[i+2]+nums[i+3]>target){
                break;
            }
            // 最大的和也不会超过target
            if((long)nums[i]+nums[len-1]+nums[len-2]+nums[len-3]<target){
                continue;
            }
            

            // 第二重循环确定第二个元素
            for(int j=i+1; j<len-2; j++){
                // 如果与前一个元素值重复，跳过
                if(j>i+1&&nums[j]==nums[j-1]){
                    continue;
                }
                // 过滤一些必然无解的情况
                // 最小的和已经超过target
                if((long)nums[i]+nums[j]+nums[j+1]+nums[j+2]>target){
                    break;
                }
                // 最大的和也不会超过target
                if((long)nums[i]+nums[len-1]+nums[len-2]+nums[j]<target){
                    continue;
                }

                // 双指针求解，从两侧开始
                int left=j+1, right=len-1;
                while(left<right){
                    int sum=nums[i]+nums[j]+nums[left]+nums[right];
                    // 满足情况，加入
                    if(sum==target){
                        // Arrays.asList快速新建List
                        result.add(Arrays.asList(nums[i], nums[j], nums[left], nums[right]));

                        // 去除重复情况
                        while(left<right&&nums[left]==nums[left+1]){
                            left++;
                        }
                        left++;
                        while(left<right&&nums[right]==nums[right-1]){
                            right--;
                        }
                        right--;
                    }
                    // 不满足情况，left++
                    else if(sum<target){
                        left++;
                    }
                    // right--
                    else if(sum>target){
                        right--;
                    }
                }
            }
        }
        return result;
    }
}

              # 两个数组的交集open in new window

              给定两个数组 nums1 和 nums2 ，返回 它们的交集 。输出结果中的每个元素一定是 唯一 的。我们可以 不考虑输出结果的顺序 。

              # 解决方案

              # 排序双指针

              class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        List<Integer> temp = new ArrayList<>();
        // 对数组进行排序
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        // 双指针选取交集
        int i=0, j=0;
        int lenS1=nums1.length, lenS2=nums2.length;
        while(i<lenS1&&j<lenS2){
            // 去除重复情况
            while(i>0&&i<lenS1&&nums1[i]==nums1[i-1]){
                i++;
            }
            while(j>0&&j<lenS2&&nums2[j]==nums2[j-1]){
                j++;
            }
            if(i>=lenS1||j>=lenS2){
                break;
            }
            // 交集加入
            if(nums1[i]==nums2[j]){
                temp.add(nums1[i]);

                // 指针移动
                i++;
                j++;
            }
            // nums1较小
            else if(nums1[i]<nums2[j]){
                i++;
            }
            else{
                j++;
            }
        }
        int[] result = new int[temp.size()];
        for(int k=0; k<temp.size(); k++){
            result[k]=temp.get(k);
        }
        return result;
    }
}
